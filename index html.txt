  <script>
    let currentQuestion = 0;
    const questions = [
      {
        question: "What day did I send you that first DM?",
        answers: ["Sunday", "Saturday"],
        correct: 0,
      },
      {
        question: "What was our very first inside joke?",
        answers: ["uFABULOUS", "Double Identities - uNasiphi noAlizwa"],
        correct: 1,
      },
      {
        question: "What do I absolutely adore the most about you?",
        answers: ["Your elbow", "Your voice and laughter"],
        correct: 1,
      },
      {
        question: "Which identity did I like first?",
        answers: ["uAlizwa", "uNasiphi"],
        correct: 0,
      },
      {
        question:
          "Do you think uLungile loves you? ... Spoiler alert: YES to all.",
        answers: ["Of course, what is there not to love?", "Not quite"],
        correct: 0,
      },
    ];

    function loadQuestion() {
      const game = document.getElementById("game");
      if (currentQuestion >= questions.length) {
        game.innerHTML = "<p>Game over! You aced it!</p>";
        return;
      }

      const q = questions[currentQuestion];
      game.innerHTML = `<p class="question">${q.question}</p>`;
      q.answers.forEach((answer, index) => {
        const button = document.createElement("button");
        button.textContent = answer;
        button.className = "answer";
        button.onclick = () => checkAnswer(index === q.correct);
        game.appendChild(button);
      });
    }

    function checkAnswer(correct) {
      const result = document.getElementById("result");
      if (correct) {
        result.textContent = "Correct! Moving to the next question...";
        currentQuestion++;
        loadQuestion();
      } else {
        result.textContent = "Not quite, try again!";
      }
    }

  </footer>

    loadQuestion();
  </script>
